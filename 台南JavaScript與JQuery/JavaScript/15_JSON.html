JSON 是陣列和物件的組合 1. 陣列放進物件裡 let temp = {屬性: []}; let friends = { GA101: [ {name: 'Andy', age: 25}, {name: 'Brian', age: 23}, {name: 'Carl', age: 21}, {name: 'Dale', age: 26}, {name: 'Ella', age: 21}, ], }; friends ---> 物件 friends.GA101 ---> 陣列 friends.GA101[1] ---> 物件 friends.GA101[1].name ---> Brian [JSON 格式] { "GA101":[ { "name":"Andy", "age":25 }, { "name":"Brian", "age":23 }, { "name":"Carl", "age":21 }, { "name":"Dale", "age":26 }, { "name":"Ella", "age":21 } ] } 2. 物件放進陣列裡 let temp = [ {}, {}, {}, {}, ]; let friends = [ {name: 'Andy', age: 25}, {name: 'Brian', age: 23}, {name: 'Carl', age: 21}, {name: 'Dale', age: 26}, {name: 'Ella', age: 21}, ]; friends ---> 陣列 friends[3] ---> 物件 friends[3].name ---> Dale [JSON 格式] [ { "name":"Andy", "age":25 }, { "name":"Brian", "age":23 }, { "name":"Carl", "age":21 }, { "name":"Dale", "age":26 }, { "name":"Ella", "age":21 } ] 3. 轉換 (1). 將 JavaScript 物件轉成 JSON JSON.stringify(JavaScript 物件) [ex] let man = { name: 'Peter', age: 40, gender: 'male', favoriateColor: ['white','black','gray'], car: { make: 'BMW', mode: 'X5', year: 2020, }, retired: false, sayHello(){ alert('Hello!') }, }; 執行: console.log(JSON.stringify(man)) 結果: {"name":"Peter","age":40,"gender":"male","favoriateColor":["white","black","gray"],"car":{"make":"BMW","mode":"X5","year":2020},"retired":false} (1). 將 JSON 轉成 JavaScript 物件(記得將 JSON 檔放在單引號或反引號中) JSON.parse(json檔) let jsonFromDB = `{"name":"Peter","age":40,"gender":"male","favoriateColor":["white","black","gray"],"car":{"make":"BMW","mode":"X5","year":2020},"retired":false}` 執行: console.log(JSON.parse(jsonFromDB)) 結果: {name: 'Peter', age: 40, gender: 'male', favoriateColor: Array(3), car: {…}, …}
